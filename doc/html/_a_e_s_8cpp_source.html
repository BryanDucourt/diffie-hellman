99  unsigned int temp;
100  const unsigned int rcon[11] = {0x00000000, 0x00000001, 0x00000002, 0x00000004,
101  0x00000008, 0x00000010, 0x00000020, 0x00000040,
102  0x00000080, 0x0000001b, 0x00000036};
103  for (int i = 0; i < 4; i++) {
104  round_key[i] = key[4 * i + 3] | (key[4 * i + 2] << 8) | (key[4 * i + 1] << 16) | (key[4 * i] << 24);
105  }

240 unsigned char AES::GaloisMultiplication(unsigned char Num_L, unsigned char Num_R)
241 {
242 //定义变量
243  unsigned char Result = 0; //伽罗瓦域内乘法计算的结果
244 
245  while (Num_L)
246  {
247 //如果Num_L最低位是1就异或Num_R，相当于加上Num_R * 1
248  if (Num_L & 0x01)
249  {
250  Result ^= Num_R;
251  }
252 
253 //Num_L右移一位，相当于除以2
254  Num_L = Num_L >> 1;
255 
256 //如果Num_R最高位为1
257  if (Num_R & 0x80)
258  {
259 //左移一位相当于乘二
260  Num_R = Num_R << 1; //注：这里会丢失最高位，但是不用担心
261 
262  Num_R ^= 0x1B; //计算伽罗瓦域内除法Num_R = Num_R / (x^8(刚好丢失最高位) + x^4 + x^3 + x^1 + 1)
263  }
264  else
265  {
266 //左移一位相当于乘二
267  Num_R = Num_R << 1;
268  }
269  }
270  return Result;
271 }

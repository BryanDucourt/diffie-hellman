198 uint16_t calc_checksum(void *pkt, int len) {
199  // 将TCP伪首部、首部、数据部分划分成16位的一个个16进制数
200  uint16_t *buf = (uint16_t *) pkt;
201  // 将校验和置为0，设置为32bit是为了保留下来16bit计算溢出的位
202  uint32_t checksum = 0;
203  // 对16位的数逐个相加，溢出的位加在最低位上
204  while (len > 1) {
205  checksum += *buf++;
206  // 前半部分将溢出的位移到最低位，后半部分去掉16bit加法溢出的位（置0）
207  checksum = (checksum >> 16) + (checksum & 0xffff);
208  len -= 2;
209  }
210  if (len) {
211  checksum += *((uint8_t *) buf); // 加上最后8位
212  checksum = (checksum >> 16) + (checksum & 0xffff);
213  }
214  return (uint16_t) ((~checksum) & 0xffff); // 取反
215 }
